# explore the temperatures 

temperatures = list(map(lambda x: x['temperature'], all_negative_submissions))
temperatures = filter_dict(lambda x: x, temperatures)

hyperthermia_c = 40
hyperthermia_f = 104 
high_temp_c = 37 

hypothermia_c = 35 
hypothermia_f = 95
high_temp_f = 98.6

correct_temps = filter_dict(lambda x: x < hyperthermia_c and x >= hypothermia_c, temperatures) + filter_dict(lambda x: x < hyperthermia_f and x >= hypothermia_f, temperatures)

high_temps = filter_dict(lambda x: x < hyperthermia_c and x >= high_temp_c, correct_temps) + filter_dict(lambda x: x < hyperthermia_f and x >= high_temp_f, correct_temps)


print("=============================================================================")
print("Number of submissions with temperature information: {}".format(len(temperatures)))
print("Number of submissions based on the hypo/hyper thermia range: {}".format(len(correct_temps)))

audio_tags = {'audio_aaa':0, 'audio_az':0, 'audio_cough':0, 'audio_count':0, 'audio_eee':0, 'audio_ooo':0}

for sub in all_negative_submissions:
    for k in audio_tags.keys():
        try:
            if len(sub[k]) > 0:
                audio_tags[k] += 1
        except:
            continue

plt.figure()
plt.bar(np.arange(len(audio_tags)), [v for k,v in audio_tags.items()])
plt.xticks(np.arange(len(audio_tags)), [k for k,v in audio_tags.items()])
plt.title('Number of attached recordings for negative covid-19\n ')
plt.axhline(y=len(all_negative_submissions),linewidth=1, color='black', linestyle='--', label='Total Number of submissions')
